SINTAX TABLE USER
CREATE TABLE users(
    id INT(11) PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(100) NOT NULL,
    pass VARCHAR(50) NOT NULL,
    name VARCHAR(255) NOT NULL,
    noHp VARCHAR(20) NOT NULL,
    email VARCHAR(255) NOT NULL);
